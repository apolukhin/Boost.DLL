Header <boost/dll/shared_library.hpp>

Contains the boost::dll::shared_library class, main class for all the DLL/DSO operations. 

namespace boost {
  namespace dll {
    class shared_library;

    // Very fast equality check that compares the actual DLL/DSO objects. Throws nothing. 
    bool operator==(const shared_library & lhs, const shared_library & rhs);

    // Very fast inequality check that compares the actual DLL/DSO objects. Throws nothing. 
    bool operator!=(const shared_library & lhs, const shared_library & rhs);

    // Compare the actual DLL/DSO objects without any guarantee to be stable between runs. Throws nothing. 
    bool operator<(const shared_library & lhs, const shared_library & rhs);

    // Swaps two shared libraries. Does not invalidate symbols and functions loaded from libraries. Throws nothing. 
    void swap(shared_library & lhs, shared_library & rhs);
  }
}

Header <boost/dll/refcountable.hpp>

Header <boost/dll/refcountable.hpp>

Includes all the reference counting functions that hold a shared pointer to the instance of boost::dll::shared_library. Those methods use boost::shared_ptr<T> and boost::function<T> classes. All those methods share reference count to an actual DLL/DSO, so it is safe to import multiple symbols from the same DLL/DSO. 

namespace boost {
  namespace dll {
    template<typename T> 
      boost::function< T > 
      import_function(const boost::shared_ptr< shared_library > &, 
                      boost::string_ref);
    template<typename T> 
      boost::function< T > 
      import_function(const boost::filesystem::path &, boost::string_ref);
    template<typename T> 
      boost::function< T > 
      import_function_alias(const boost::shared_ptr< shared_library > &, 
                            boost::string_ref);
    template<typename T> 
      boost::function< T > 
      import_function_alias(const boost::filesystem::path &, 
                            boost::string_ref);
    template<typename T> 
      boost::shared_ptr< T > 
      import_variable(const boost::shared_ptr< shared_library > &, 
                      boost::string_ref);
    template<typename T> 
      boost::shared_ptr< T > 
      import_variable(const boost::filesystem::path &, boost::string_ref);
    template<typename T> 
      boost::shared_ptr< T > 
      import_variable_alias(const boost::shared_ptr< shared_library > &, 
                            boost::string_ref);
    template<typename T> 
      boost::shared_ptr< T > 
      import_variable_alias(const boost::filesystem::path &, 
                            boost::string_ref);
  }
}
